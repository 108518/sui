dialog(bootstrap中命名为Modal)


无js触发,确定按钮对应html添加data-ok="modal"属性

data-属性都是必须的。也可通过$("xx").modal(opt)触发,参数可参考 bootstrap2.3.2#Modal

涉及弹层html结构的参数（如hasfoot，okbtn，cancelbtn）不适用此类调用方法，因为弹层html结构完全是可控的，比如不要脚部，则.modal-footer不写到html里即可

.modal 改为 .sui-modal | .modal-backdrop 改为 .sui-modal-backdrop


js触发,静态方法$.alert(opt) $.confirm(opt) ,以及原型方法$('xx').modal(opt)

jquery弹层静态方法，用于很少重复，不需记住状态的弹层，可方便的直接调用，最简单形式就是$.alert('我是alert')

若弹层内容是复杂的Dom结构， 建议将弹层html结构写到模版里，用$(xx).modal(options) 调用

/* 
* 
* example
* $.alert({
*  backdrop: true   true/false表示蒙层是黑／白 半透明
*  keyboard: true   是否可由esc按键关闭
*  title: '自定义标题'
*  body: 'html' //必填

*  cancelBtn : '雅达'
*  hasfoot: {Boolean}  是否显示脚部  默认true
*  width: {number|string(px)|'small'|'normal'|'large'}推荐优先使用后三个描述性字符串，统一样式
*  timeout: {number} 1000    单位毫秒ms ,dialog打开后多久自动关闭
*  show:     fn --------------function(e){}
*  shown:    fn
*  hide:     fn
*  hidden:   fn
*  okHide:   function(e){alert('点击确认后、dialog消失前的逻辑,
*            函数返回true（默认）则dialog关闭，反之不关闭;若不传入则默认是直接返回true的函数
*            注意不要人肉返回undefined！！')}
*  okHidden: function(e){alert('点击确认后、dialog消失后的逻辑')}
* })
*/

$('#J_confirm').on('click', function(e){
  $.confirm({
    body: 'confirm again'
    ,width: 'normal'
    ,backdrop: false
    ,timeout: 1000
    ,show: function(){console.log('show')}
    ,shown: function(){console.log('shown')}
    ,hide: function(){console.log('hide')}
    ,hidden: function(){console.log('hiden')}
    ,okHide: function(){var a=confirm('true or false');if(!a) return false}
    ,okHidden: function(){console.log('okHidden')}
  })
})
